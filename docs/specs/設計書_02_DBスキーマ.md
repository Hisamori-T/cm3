```sql
-- インデックス
CREATE INDEX ix_payments_invoice_id ON payments(invoice_id);

-- 案件単位の請求・入金集計ビュー
CREATE VIEW project_invoice_summary AS
SELECT
  i.project_id,
  COUNT(i.id) AS invoice_count,
  COALESCE(SUM(i.total_amount), 0) AS total_billed,
  COALESCE(SUM(p.paid), 0) AS total_paid,
  COALESCE(SUM(i.total_amount), 0) - COALESCE(SUM(p.paid), 0) AS outstanding,
  MAX(i.payment_due_date) AS latest_due_date
FROM invoices i
LEFT JOIN (SELECT invoice_id, SUM(amount) AS paid FROM payments GROUP BY invoice_id) p
  ON p.invoice_id = i.id
WHERE i.status != 'cancelled'
GROUP BY i.project_id;
```

### 3.9 QCDS 計算ロジック

```python
# services/qcds_calculator.py より

def calculate_qcds(qcds, direct_works, project_price):
    pp = project_price or 0.0
    
    # A: 直接工事費
    direct_cost_budget     = SUM(w.budget_amount)
    direct_cost_agreed     = SUM(w.agreed_amount)
    direct_cost_settlement = SUM(w.settlement_amount)
    
    # B-1: 保険料
    labor_insurance       = round(pp × labor_insurance_rate)
    construction_insurance = round(pp × 1.1 × construction_insurance_rate)
    
    # B-2: 現場担当者給与
    site_personnel_cost = actual_site_personnel_cost or round(pp × site_staff_salary_rate)
    
    # B-3: 固定費（事務用品+通信交通+雑費+予備費+産廃）
    fixed_overhead = SUM(office_supplies, communication_cost, misc_cost, ...)
    
    # B 合計
    site_overhead_total = labor_insurance + construction_insurance + ... + fixed_overhead
    
    # D: 工事部経費
    construction_dept_overhead = round(pp × common_overhead_rate)
    
    # G: 一般管理費
    general_admin_cost = round(pp × general_admin_rate)
    
    # H: 営業利益
    total_cost = (agreed or budget) + site_overhead_total + construction_dept_overhead + shared_overhead
    operating_profit = pp - total_cost - general_admin_cost
```
